44 /**
45  * The parsers keeps track of their states with the states defined here, and
46  * use these symbols when pushing to the stack so that they can later recreate
47  * their environments when re-entered.
48  */
49 typedef enum {
50  /* Universal states. */
51  STATE_INITIAL, /**< This is the initial state for every parser. */
52  STATE_COMPLETE, /**< Complete! */
53  STATE_PARSE_TAG,
54  STATE_PARSE_TAG_DONE,
55  /* ParseHTML states. */
56  STATE_PRE_HEAD, /**< In this state, we've not detected head yet. */
57  STATE_PRE_BODY, /**< In this state, we'll consider frames vs. body. */
58  STATE_PARSE_BODY, /**< In this state, we can parse the body. */
59  STATE_PARSE_HEAD, /**< In this state, we will setup head for parsing. */
60  STATE_PARSE_HEAD_DONE, /**< Resume here after parsing head. */
61  STATE_PARSE_NOFRAMES, /**< In this state, we can parse noframes content. */
62  STATE_PARSE_NOFRAMES_DONE, /**< In this state, we can restore more state. */
63  STATE_PARSE_FRAMESET, /**< In this state, we will parse frameset content. */
64  STATE_PARSE_FRAMESET_DONE, /**< We need to cleanup some things after parsing frameset. */
65 } parserState;

68 /**
69  * This typedef represents the state of a parser when it enters and exits.
70  * When the parser needs to finish work on the way back up the stack, it will
71  * push one of these records to the stack, and it will pop a record from the

79  GetTokenMode reentry_mode; /**< The mode to use for the next node. */
80  parserState reentry_state; /**< State to set during re-entry. */
81  GetTokenMode mode; /**< The caller will peek at this value to get the correct mode. */
